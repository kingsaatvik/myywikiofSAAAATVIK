  <nav>

    <ul>

      <li><a onclick="showPage('school')">School</a></li>

      <li><a onclick="showPage('hobbies')">Hobbies</a></li>

      <li><a onclick="showPage('family')">Family</a></li>

      <li><a onclick="showPage('work')">Work</a></li>

    </ul>

  <div id="content">

    <p>Select a topic above.</p>

  </div>


  <h2>Edit My Journal</h2>

  <textarea id="editor" rows="5" placeholder="Write about your day, thoughts, or anything you want..."></textarea>

  <div class="controls">

    <button onclick="publish()">Publish</button>

    <button onclick="downloadEntries()">Download Entries</button>

    <button onclick="clearAll()">Clear All Entries</button>

  </div>


  <div id="entries">

    <h3 class="small">Saved Entries</h3>

    <div id="entriesList"><p class="small">No entries yet.</p></div>

  </div>


  <script>

    const pages = {

      school: "I study at The Heritage School Rohini and love mathematics.",

      hobbies: "I enjoy cycling, coding, and playing.",

      family: "I live with my parents and my grandfather + my big sister.",

      work: "I want to be a surgeon someday!"

    };


    const STORAGE_KEY = 'mywiki_entries_v1';


    window.addEventListener('DOMContentLoaded', () => {

      renderEntries();

      const savedView = localStorage.getItem('mywiki_last_view');

      if (savedView && pages[savedView]) showPage(savedView);

    });


    function showPage(page) {

      const content = document.getElementById('content');

      content.innerHTML = "<h2>" + page.charAt(0).toUpperCase() + page.slice(1) + "</h2><p>" + pages[page] + "</p>";

      localStorage.setItem('mywiki_last_view', page);

    }


    function publish() {

      const editor = document.getElementById('editor');

      const text = editor.value.trim();

      if (text === "") return;

      const entries = loadEntries();

      const entry = { text: text, time: new Date().toISOString() };

      entries.unshift(entry);

      localStorage.setItem(STORAGE_KEY, JSON.stringify(entries));

      editor.value = "";

      renderEntries();

      document.getElementById('content').innerHTML = "<h2>My Journal Entry</h2><p>" + escapeHtml(text) + "</p>";

    }


    function loadEntries() {

      try {

        const raw = localStorage.getItem(STORAGE_KEY);

        return raw ? JSON.parse(raw) : [];

      } catch (e) {

        return [];

      }

    }


    function renderEntries() {

      const list = document.getElementById('entriesList');

      const entries = loadEntries();

      if (!entries.length) {

        list.innerHTML = '<p class="small">No entries yet.</p>';

        return;

      }

      list.innerHTML = entries.map((e) => {

        const time = new Date(e.time).toLocaleString();

        return '<div class="entry"><div class="small">' + time + '</div><div>' + escapeHtml(e.text) + '</div></div>';

      }).join('');

    }


    function clearAll() {

      if (!confirm('Clear all saved journal entries?')) return;

      localStorage.removeItem(STORAGE_KEY);

      renderEntries();

      document.getElementById('content').innerHTML = '<p>Select a topic above.</p>';

    }


    function downloadEntries() {

      const entries = loadEntries();

      if (!entries.length) return alert('No entries to download.');

      const text = entries.map(e => new Date(e.time).toLocaleString() + "\n" + e.text + "\n\n").join('');

      const blob = new Blob([text], { type: 'text/plain' });

      const url = URL.createObjectURL(blob);

      const a = document.createElement('a');

      a.href = url;

      a.download = 'mywiki_entries.txt';

      document.body.appendChild(a);

      a.click();

      a.remove();

      URL.revokeObjectURL(url);

    }


    function escapeHtml(str) {

      return str.replace(/[&<>"']/g, function(m) {

        return ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'})[m];

      });

    }
